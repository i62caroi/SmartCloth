177void getAcumuladoHoyFromSD(){
178
179 char *today = rtc.getDateStr(); // Es posible que se cambie de día durante el cocinado?? Si no, hacer en setupRTC().
180
181 // SUMAS
182 float sumCarb = 0.0, sumLip = 0.0, sumProt = 0.0, sumKcal = 0.0, sumPeso = 0.0;
183 int sumCarb_R = 0, sumLip_R = 0, sumProt_R = 0;
184 int nComidas = 0;
185
186 char lineBuffer[128];
187
188 char *token;
189 int fieldIndex;
190
191 bool msg = true;
192
193
194 myFile = SD.open(fileCSV, FILE_READ);
195 if (myFile){
196 while (myFile.available()) {
197
198 myFile.readBytesUntil('\n', lineBuffer, sizeof(lineBuffer) - 1); // Leer línea completa hasta el tamaño máximo del búfer
199 lineBuffer[sizeof(lineBuffer) - 1] = '\0'; // Asegurar terminación nula
200
201 //Serial.println(lineBuffer);
202
203 token = strtok(lineBuffer, ";"); // Separar campos de la línea utilizando el delimitador ';'
204 fieldIndex = 0;
205
206 if(strcmp(today, token) == 0){ // today == primer token ==> comida guardada hoy
207
208 if(msg){
209 Serial.println(F("Obteniendo Acumulado Hoy..."));
210 msg = false; // Solo imprimir una vez y si hay algo que sumar
211 }
212
213 nComidas++; // Incrementar numero comidas guardadas hoy
214
215 float valueFloat;
216 int valueInt;
217
218 while (token != NULL) {
219
220 switch (fieldIndex){ // fieldIndex = 0 => fecha fieldIndex = 1 => hora
221 case 2: valueFloat = atof(token); sumCarb += valueFloat; break; // Carbohidratos
222 case 3: valueInt = atoi(token); sumCarb_R += valueInt; break; // Raciones de carbohidratos
223 case 4: valueFloat = atof(token); sumLip += valueFloat; break; // Lípidos (Grasas)
224 case 5: valueInt = atoi(token); sumLip_R += valueInt; break; // Raciones de lípidos
225 case 6: valueFloat = atof(token); sumProt += valueFloat; break; // Proteínas
226 case 7: valueInt = atoi(token); sumProt_R += valueInt; break; // Raciones de proteínas
227 case 8: valueFloat = atof(token); sumKcal += valueFloat; break; // Kilocalorías
228 case 9: valueFloat = atof(token); sumPeso += valueFloat; break; // Peso
229 }
230
231 // Obtener el siguiente campo
232 token = strtok(NULL, ";");
233 fieldIndex++;
234 }
235 }
236
237 }
238
239 myFile.close();
240
241
242 // ----- ACTUALIZAR ACUMULADO HOY -----
243 ValoresNutricionales valAux(sumCarb, sumLip, sumProt, sumKcal);
244 diaActual.setValoresDiario(valAux); // Inicializar valores nutricionales del Acumulado Hoy
245 diaActual.setPesoDiario(sumPeso); // Actualizar peso del Acumulado Hoy
246 diaActual.setNumComidas(nComidas); // Actualizar nº de comidas del Acumulado Hoy
247
248 }
249 else{
250 Serial.println("Error opening file for reading!");
251 }
252
253}
