102 // debe forzar la liberación (eventoBascula = LIBERAR) para, tras guardar o no la comida (STATE_saved),
103 // o borrar o no plato (STATE_deleted), se pueda regresar a INI.

105 // cambios en el peso (diffWeight > 2.0).

109 addEventToBuffer(eventoBascula);
110 flagEvent = true;
111 }*/ // ------------------------------------------------------------------
112
113
114 lastWeight = newWeight;
115 newWeight = weight;
116 diffWeight = abs(lastWeight - newWeight);
117 if (diffWeight > 2.0){ // Cambio no causado por "interferencias" de la báscula
118
119 /* 'pesoBascula' representa el peso evitando pequeños saltos en las medidas.
120 Este valor es el que se usará como peso de los alimentos.
121 Hasta ahora se usaba 'weight' directamente, pero al añadir un nuevo alimento
122 al plato se tomaba el último valor de 'weight', que podría haber variado algo
123 respecto a lo que se mostraba por sucesivas medidas de la báscula. */
124
125 pesoBascula = newWeight;
126 if(pesoBascula < 1.0) pesoBascula = 0.0; //fixPesoBascula
127
128 if(lastWeight < newWeight){ //INCREMENTO
129 Serial.println(F("\nIncremento..."));
130 if(lastWeight > -1.0){ //Incremento en positivo (salvando valores cercanos a 0)
131 Serial.print(F("\nINCREMENTO"));
132 eventoBascula = INCREMENTO;
133 }
134 else if(lastWeight <= -1.0){ //Incremento por tara viniendo de negativo
135 Serial.print(F("\nTARADO"));
136 eventoBascula = TARAR;
137 }
138 }
139 else{ //DECREMENTO
140 Serial.println(F("\nDecremento..."));
141 if(newWeight >= 1.0){ //Se ha quitado algo pero no todo (sigue positivo)
142 Serial.print(F("\nDECREMENTO"));
143 eventoBascula = DECREMENTO;
144 }
145 else if(newWeight < 1.0){ //peso negativo o cercano a 0.0
146 if(tarado){ //Decremento debido a tara
147 Serial.print(F("\nTARADO"));
148 eventoBascula = TARAR;
149 }
150 else{ //Decremento por retirar objeto
151 if(abs(abs(newWeight) - pesoARetirar) < 5.0){ //Nuevo peso (negativo) es contrario (-X = +X) al peso del plato + recipiente ==> se ha quitado todo
152 // Se ha puesto un umbral de 5 gr para saber si se ha retirado todo, pero podría reducirse a 1 gr
153 Serial.print(F("\nLIBERADA"));
154 eventoBascula = LIBERAR;
155 }
156 else{ //Se están quitando elementos de la báscula tras haber tarado, por eso se baja a negativo. Se pasa por 'QUITAR' antes de 'LIBERAR'.
157 Serial.print(F("\nQUITANDO"));
158 eventoBascula = QUITAR;
159 }
160 }
161 }
162 }
163
164 Serial.println(F("\n--------------------------------------"));
165 Serial.print(F("\nPeso anterior: ")); Serial.println(lastWeight);

167 Serial.print(F("Peso Bascula: ")); Serial.println(pesoBascula);
