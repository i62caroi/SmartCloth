204void Comida::updateValoresComida(bool suma, ValoresNutricionales val){
205 float carb, lip, prot, kcal;
206 if(suma){ // Añadir plato
207 carb = _valoresComida.getCarbValores() + val.getCarbValores();
208 lip = _valoresComida.getLipValores() + val.getLipValores();
209 prot = _valoresComida.getProtValores() + val.getProtValores();
210 kcal = _valoresComida.getKcalValores() + val.getKcalValores();
211 }
212 else{ // Eliminar plato
213 carb = _valoresComida.getCarbValores() - val.getCarbValores();
214 lip = _valoresComida.getLipValores() - val.getLipValores();
215 prot = _valoresComida.getProtValores() - val.getProtValores();
216 kcal = _valoresComida.getKcalValores() - val.getKcalValores();
217 }
218 ValoresNutricionales valAux(carb, lip, prot, kcal);
219 _valoresComida.setValores(valAux);
220}
