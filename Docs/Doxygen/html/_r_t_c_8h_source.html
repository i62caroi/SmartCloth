194uint8_t getLastSunday(uint8_t month, uint16_t year) {
195 // Obtener el día máximo del mes
196 uint8_t lastDay = 31;
197 if (month == 4 || month == 6 || month == 9 || month == 11) { // Abril, Junio, Septiembre y Noviembre
198 lastDay = 30;
199 } else if (month == 2) { // Febrero
200 if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)) { // Año bisiesto
201 lastDay = 29;
202 } else {
203 lastDay = 28;
204 }
205 }
206
207 // Obtener el DOW del último día del mes
208 uint8_t lastDOW = getDOW(lastDay, month, year);
209
210 // Calcular el último domingo restando el DOW del último día al día máximo del mes:
211 // Por ejemplo, si el día 31 de marzo es jueves (día 4), entonces 31-4 = 27. El último
212 // domingo es el día 27.
213 // En el caso en que el último día del mes (p.ej. 31) caiga en Domingo, entonces se
214 // devuelve ese día (31).
215 if (lastDOW == 7) return lastDay;
216 else return lastDay - lastDOW;
217
218}

238uint8_t getDOW(uint8_t day, uint8_t month, uint16_t year) {
239 // Ajustar Enero y Febrero como meses 13 y 14 del año anterior. Necesario para el algoritmo.
240 if (month < 3) {
241 month += 12;
242 year--;
243 }
244 //else month = month - 2;
245
246 //uint8_t century = year / 100 + 1;
247 uint8_t century = 21; // Queda siglo 21 para largo
248 uint8_t yearOfCentury = year % 100;
249
250 // Fórmula del Algoritmo de Zeller
251 uint8_t dow = (day + ((13 * (month + 1))/5) + yearOfCentury + (yearOfCentury/4) + (century/4) - (2*century)) % 7;
252
253 return dow + 1; // Pasar de L(0) - D(6) a L(1) - D(7)
254}
